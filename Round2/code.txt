import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class CopyFile {
    public static void main(String[] args) {
        try {
            // Open input stream to read from this Java file
            BufferedReader reader = new BufferedReader(new FileReader("CopyFile.java"));
            
            // Open output stream to write to a new file
            BufferedWriter writer = new BufferedWriter(new FileWriter("code.txt"));
            
            // Read and write line by line
            String line;
            while ((line = reader.readLine()) != null) {
                writer.write(line);
                writer.newLine(); // Add newline character
            }
            
            // Close streams
            reader.close();
            writer.close();
            
            System.out.println("File copied successfully!");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
